    public int minArray(int[] numbers) {
        for(int i=0;i<numbers.length-1;i++)

            if(numbers[i]>numbers[i+1])
                return numbers[i+1];
        }
        return numbers[0];
    }

class Solution {
    boolean[][] visit;
    private int[][] dir= {{-1,0},{0,-1},{0,1},{1,0}};
    public boolean exist(char[][] board, String word) {
        int rows = board.length;
        int cols = board[0].length;
        visit = new boolean[rows][cols];
        for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
                if (word.charAt(0) == board[i][j] && dfs(board, word, i, j, 0)) {
					return true;
				}
			}
		}
		return false;
    }
    private boolean dfs(char[][] board, String word, int i, int j, int index) {
		if (index == word.length()) {
			return true;
		}
		if (i >= board.length || i < 0 || j >= board[0].length || j < 0 || visit[i][j] || word.charAt(index) != board[i][j]) return false;
		visit[i][j] = true;
		for (int[] d : dir) {
			if (dfs(board, word, i + d[0], j + d[1], index + 1))  return true;
		}
		visit[i][j] = false;
		return false;
	}
}
