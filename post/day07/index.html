class Solution {
    public double myPow(double x, int n) {
        if(n==-1) return 1/x;
        if(n==0) return 1;
        double result=1;
        if(n<0){
            for(int i=1;i<=-n;i++){
                result*=x;
            }
            return 1/result;
        }else{
            for(int i=1;i<=n;i++){
                result*=x;
            }
            return result;
        }
    }
}

class Solution {
    public double myPow(double x, int n) {
        if(n==-1) return 1/x;
        if(n==0||x==1) return 1;
        double num=1;
        long N=n;
        if(N<0){
            N=-N;
            x=1/x;
        }
        while(N>0){
            if(N%2==1) num=num*x;
            N=N>>1;
            x=x*x;
        }
        return num;
    }
